public class Education {
  public static void main(String[] args) {
    Degree graduation = new Degree(
      "Análise e desenvolvimento de sistemas",
      "Faculdade Anhanguera",
      "Anápolis-GO",
      "2018 – 2021"
    );
    
    Degree postgrad = new Degree(
      "Pós-graduação em engenharia de software",
      "Faculdade Descomplica",
      "Remoto",
      "Cursando - conclusão em 01/2026"
    );
    
    System.out.println("FORMAÇÃO ACADÊMICA\n");
    graduation.printDetails();
    System.out.println();
    postgrad.printDetails();
  }
}

class Degree {
  String title;
  String institution;
  String location;
  String period;
  
  Degree(String title, String institution, String location, String period) {
    this.title = title;
    this.institution = institution;
    this.location = location;
    this.period = period;
  }
  
  void printDetails() {
    System.out.println("* " + title);
    System.out.println("  " + institution + " | " + location);
    System.out.println("  " + period);
  }
}

public class Skills {
  public static void main(String[] args) {
    DeveloperSkills skills = new DeveloperSkills();
    skills.printAllSkills();
  }
}

class DeveloperSkills {
  String[] languages = { "Java (7/8+)", "SQL", "JavaScript", "Flutter/Dart" };
  String[] frameworks = { "Spring Boot", "JSF", "Swing", "RESTful APIs", "Node.js", "React Native" };
  String[] databases = { "Oracle", "MySQL", "PostgreSQL", "SQLite" };
  String[] devops = { "AWS", "Azure", "DevOps", "Docker", "CI/CD" };
  String[] methodologies = { "Scrum", "Metodologias Ágeis", "TDD", "OOP" };
  String[] tools = { "Gradle", "Git", "Jira", "Reporting" };
  String[] softSkills = { "Resolução de Problemas", "Comunicação", "Trabalho em Equipe", "Adaptabilidade" };
  
  void printAllSkills() {
    System.out.println("HABILIDADES TÉCNICAS E COMPORTAMENTAIS\n");
    
    printSkillCategory("Linguagens", languages);
    printSkillCategory("Frameworks/Bibliotecas", frameworks);
    printSkillCategory("Bancos de Dados", databases);
    printSkillCategory("Nuvem/DevOps", devops);
    printSkillCategory("Metodologias", methodologies);
    printSkillCategory("Ferramentas", tools);
    printSkillCategory("Soft Skills", softSkills);
    
    System.out.println("\nIDIOMAS");
    System.out.println("* Português (Nativo)");
    System.out.println("* Inglês (Intermediário)");
  }
  
  void printSkillCategory(String category, String[] skills) {
    System.out.println("* " + category + ": " + String.join(", ", skills));
  }
}
